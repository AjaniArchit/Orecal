 --Assignment operator(:=) 

Declare
    A Integer:=10;
    B Integer:=20;
    C Integer;
    F Real;

Begin
     C:=A+B;
     Dbms_output.put_line('value of C:'||C);
     
     F:=70.0/3.0;
     Dbms_output.put_line('value of F:'||F);

End;

[OUTPUT]

  value of C:30 
  value of F:23.33333333333333333333333333333333333333
